020package org.apache.directory.server.dhcp.messages;
021
022import com.google.common.base.CharMatcher;
023import com.google.common.base.Splitter;
024import com.google.common.collect.Iterables;
025import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
026import java.beans.PropertyEditorSupport;
027import javax.annotation.Nonnull;
028
029/**
030 * A representation of a DHCP hardware address.
031 *
032 * @author <a href="mailto:dev@directory.apache.org">Apache Directory Project</a>
033 */
034public final class HardwareAddress {
035
036    public static class PropertyEditor extends PropertyEditorSupport {
037
038        @Override
039        public String getJavaInitializationString() {
040            String text = getAsText();
041            if (text == null)
042                return "null";
043            else
044                return "HardwareAddress.fromString(\"" + text + "\")";
045        }
046
047        @Override
048        public void setAsText(String text) {
049            if (text == null)
050                setValue(null);
051            else
052                setValue(HardwareAddress.fromString(text));
053        }
054    }
055    /**
056     * [htype] Hardware address type, see ARP section in "Assigned Numbers" RFC;
057     * e.g., '1' = 10mb ethernet.
058     */
059    private final short type;
060    /**
061     * [hlen] Hardware address length (e.g. '6' for 10mb ethernet).
062     */
063    private final short length;
064    /**
065     * [chaddr] Client hardware address.
066     */
067    private final byte[] address;
068
069    /**
070     * @param type
071     * @param length
072     * @param address
073     */
074    @SuppressFBWarnings("EI_EXPOSE_REP2")
075    public HardwareAddress(short type, short length, @Nonnull byte[] address) {
076        this.type = type;
077        this.length = length;
078        this.address = address;
079    }
080
081    public HardwareAddress(@Nonnull HardwareAddressType type, short length, @Nonnull byte[] address) {
082        this(type.getCode(), length, address);
083    }
084
085    public HardwareAddress(@Nonnull HardwareAddressType type, @Nonnull byte[] address) {
086        this(type.getCode(), (short) address.length, address);
087    }
088
089    @Nonnull
090    @SuppressFBWarnings("EI_EXPOSE_REP")
091    public byte[] getAddress() {
092        return address;
093    }
094
095    public short getType() {
096        return type;
097    }
098
099    public short getLength() {
100        return length;
101    }
102
103    /**
104     * @see java.lang.Object#hashCode()
105     * @return the instance's hash code
106     */
107    @Override
108    public int hashCode() {
109        int hashCode = 98643532 ^ type ^ length;
110
111        for (int i = 0; i < length; i++) {
112            hashCode ^= address[i];
113        }
114
115        return hashCode;
116    }
117
118
119    /*
120     * @see java.lang.Object#equals(java.lang.Object)
121     */
122    @Override
123    public boolean equals(Object obj) {
124        if (this == obj)
125            return true;
126        if (null == obj)
127            return false;
128        if (!getClass().equals(obj.getClass()))
129            return false;
130
131        HardwareAddress hw = (HardwareAddress) obj;
132
133        if (type != hw.type)
134            return false;
135        if (length != hw.length)
136            return false;
137        for (int i = 0; i < length; i++)
138            if (address[i] != hw.address[i])
139                return false;
140        return true;
141    }
142
143    /**
144     * Create the string representation of the hardware address native to the
145     * corresponding address type. This method currently supports only type
146     * 1==ethernet with the representation <code>a1:a2:a3:a4:a5:a6</code>.<br>
147     * For all other types, this method falls back to the representation created
148     * by toString().
149     *
150     * @see java.lang.Object#toString()
151     */
152    @Nonnull
153    public String getNativeRepresentation() {
154        StringBuilder sb = new StringBuilder();
155
156        switch (type) {
157            case 1:
158                for (int i = 0; i < length; i++) {
159                    if (i > 0) {
160                        sb.append(":");
161                    }
162
163                    String hex = Integer.toHexString(address[i] & 0xff);
164
165                    if (hex.length() < 2) {
166                        sb.append('0');
167                    }
168
169                    sb.append(hex);
170                }
171
172                break;
173
174            default:
175                sb.append(toString());
176        }
177
178        return sb.toString();
179    }
180
181    /**
182     * Create a string representation of the hardware address. The string
183     * representation is in the format<br>
184     * <code>t/a1:a2:a3...</code><br>
185     * Where <code>t</code> represents the address type (decimal) and
186     * <code>a<sub>n</sub></code> represent the address bytes (hexadecimal).
187     *
188     * @see java.lang.Object#toString()
189     */
190    @Override
191    public String toString() {
192        StringBuilder sb = new StringBuilder();
193        sb.append(type);
194        sb.append("/");
195
196        for (int i = 0; i < length; i++) {
197            if (i > 0) {
198                sb.append(":");
199            }
200
201            String hex = Integer.toHexString(address[i] & 0xff);
202
203            if (hex.length() < 2) {
204                sb.append('0');
205            }
206
207            sb.append(hex);
208        }
209
210        return sb.toString();
211    }
212
213    /**
214     * Parses a string representation of a hardware address.
215     * Valid: 1/11:22:33:44:55:66 (toString())
216     * Valid: Ethernet/11:22:33:44:55:66
217     * Valid: 11:22:33:44:55:66 (defaults to Ethernet)
218     *
219     * @param text
220     * @return HardwareAddress
221     */
222    @Nonnull
223    public static HardwareAddress fromString(@Nonnull String text) {
224        int idx = text.indexOf('/');
225        HardwareAddressType hardwareAddressType = HardwareAddressType.Ethernet;
226        if (idx != -1) {
227            String hardwareAddressTypeText = text.substring(0, idx);
228            try {
229                int hardwareAddressTypeCode = Integer.parseInt(hardwareAddressTypeText);
230                hardwareAddressType = HardwareAddressType.forCode(hardwareAddressTypeCode);
231            } catch (NumberFormatException e) {
232                // This will throw IllegalArgumentException, which is roughly what we want.
233                hardwareAddressType = HardwareAddressType.valueOf(hardwareAddressTypeText);
234            }
235            text = text.substring(idx + 1);
236        }
237
238        CharMatcher separator = CharMatcher.BREAKING_WHITESPACE.or(CharMatcher.anyOf(":-"));
239        Iterable<String> parts = Splitter.on(separator).omitEmptyStrings().trimResults().split(text);
240        int i = 0;
241        byte[] out = new byte[Iterables.size(parts)];
242        for (String part : parts)
243            out[i++] = (byte) Integer.parseInt(part, 16);
244        return new HardwareAddress(hardwareAddressType.getCode(), (short) out.length, out);
245    }
