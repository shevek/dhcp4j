005package org.anarres.dhcp.v6.io;
006
007import com.google.common.primitives.Shorts;
008import java.nio.ByteBuffer;
009import javax.annotation.Nonnull;
010import org.anarres.dhcp.v6.messages.Dhcp6Message;
011import org.anarres.dhcp.v6.messages.Dhcp6RelayMessage;
012import org.anarres.dhcp.v6.options.Dhcp6Option;
013import org.anarres.dhcp.v6.options.Dhcp6Options;
014
015/**
016 * http://tools.ietf.org/html/rfc3315#section-22.1
017 *
018 * @author shevek, marosmars
019 */
020public class Dhcp6MessageEncoder {
021
022    private static class Inner {
023
024        private static final Dhcp6MessageEncoder INSTANCE = new Dhcp6MessageEncoder();
025    }
026
027    @Nonnull
028    public static Dhcp6MessageEncoder getInstance() {
029        return Inner.INSTANCE;
030    }
031
032    /**
033     * https://tools.ietf.org/html/rfc3315#section-6
034     * https://tools.ietf.org/html/rfc3315#section-7
035     */
036    public void encode(ByteBuffer byteBuffer, Dhcp6Message message) {
037        byteBuffer.put(message.getMessageType().getCode());
038        if(Dhcp6MessageDecoder.isRelayedMessage(message.getMessageType())) {
039            final Dhcp6RelayMessage relayedMessage = (Dhcp6RelayMessage) message;
040            byteBuffer.put(relayedMessage.getHopCount());
041            byteBuffer.put(relayedMessage.getLinkAddress().getAddress());
042            byteBuffer.put(relayedMessage.getPeerAddress().getAddress());
043        } else {
044            int transactionId = message.getTransactionId();
045            byteBuffer.put((byte) ((transactionId >> 16) & 0xFF));
046            byteBuffer.put((byte) ((transactionId >> 8) & 0xFF));
047            byteBuffer.put((byte) (transactionId & 0xFF));
048        }
049        encode(byteBuffer, message.getOptions());
050    }
051
052    public void encode(@Nonnull ByteBuffer byteBuffer, @Nonnull Dhcp6Options options) {
053        for (Dhcp6Option option : options) {
054            // Option continuation per RFC3396
055            encode(byteBuffer, option);
056        }
057    }
058
059    public ByteBuffer encode(@Nonnull Dhcp6Options options) {
060        final ByteBuffer allocate = ByteBuffer.allocate(options.getLength());
061        encode(allocate, options);
062        return allocate;
063    }
064
065    public void encode(final @Nonnull ByteBuffer byteBuffer, @Nonnull final Dhcp6Option option) {
066        short tag = option.getTag();
067        byte[] data = option.getData();
068        byteBuffer.putShort(tag);
069        byteBuffer.putShort(Shorts.checkedCast(data.length));
070        byteBuffer.put(data);
071    }
072}
