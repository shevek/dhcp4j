005package org.anarres.dhcp.v6.options;
006
007import com.google.common.io.BaseEncoding;
008import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
009import java.util.Arrays;
010import javax.annotation.Nonnull;
011
012/**
013 *
014 * @author shevek
015 */
016public abstract class DuidOption extends Dhcp6Option {
017
018    @Nonnull
019    public Duid getDuid() {
020        return new Duid(getData());
021    }
022
023    public void setDuid(@Nonnull byte[] duid) {
024        setData(duid);
025    }
026
027    public void setDuid(@Nonnull Duid duid) {
028        setData(duid.getData());
029    }
030
031    public static final class Duid {
032
033        private byte[] data;
034
035        @SuppressFBWarnings("EI_EXPOSE_REP2")
036        public Duid(final byte[] data) {
037            this.data = data;
038        }
039
040        @SuppressFBWarnings("EI_EXPOSE_REP")
041        public byte[] getData() {
042            return data;
043        }
044
045        @Override
046        public int hashCode() {
047            return Arrays.hashCode(data);
048        }
049
050        @Override
051        public boolean equals(final Object o) {
052            if (this == o)
053                return true;
054            if (o == null || getClass() != o.getClass())
055                return false;
056            final Duid duid = (Duid) o;
057            return Arrays.equals(data, duid.data);
058        }
059
060        @Override
061        public String toString() {
062            return "Duid{"
063                    + "data=" + BaseEncoding.base16().encode(data)
064                    + '}';
065        }
066    }
067}
