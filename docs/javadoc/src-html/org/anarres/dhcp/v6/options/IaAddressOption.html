005package org.anarres.dhcp.v6.options;
006
007import com.google.common.base.Optional;
008import com.google.common.base.Preconditions;
009import java.net.Inet6Address;
010import java.net.InetAddress;
011import java.net.UnknownHostException;
012import java.nio.ByteBuffer;
013import javax.annotation.Nonnull;
014import org.anarres.dhcp.v6.io.Dhcp6MessageEncoder;
015import org.apache.directory.server.dhcp.DhcpException;
016
017/**
018 * https://tools.ietf.org/html/rfc3315#section-22.6
019 *
020 * @author marosmars
021 */
022public class IaAddressOption extends SuboptionOption {
023
024    private static final short TAG = 5;
025    private static final int HEADER_SIZE = 24;
026
027    @Override
028    public short getTag() {
029        return TAG;
030    }
031
032    @Nonnull
033    public InetAddress getIp() {
034        ByteBuffer buf = ByteBuffer.wrap(getData());
035        final byte[] dst = new byte[16];
036        buf.get(dst, 0, 16);
037
038        try {
039            return Inet6Address.getByAddress(dst);
040        } catch (UnknownHostException e) {
041            // This can happen only if the IP byte array is of illegal length
042            throw new IllegalStateException("Illegal IP address", e);
043        }
044    }
045
046    public void setIp(@Nonnull final InetAddress address) {
047        Preconditions.checkArgument(address instanceof Inet6Address, "Ipv4 detected: %s", address);
048        ByteBuffer buf = ByteBuffer.wrap(getData());
049        buf.put(address.getAddress());
050    }
051
052    public void setPreferredLifetime(int preferredLifetime) {
053        ByteBuffer buf = ByteBuffer.wrap(getData());
054        buf.putInt(16, preferredLifetime);
055    }
056
057    public int getPreferredLifetime() {
058        ByteBuffer buf = ByteBuffer.wrap(getData());
059        return buf.getInt(16);
060    }
061
062    public int getValidLifetime() {
063        ByteBuffer buf = ByteBuffer.wrap(getData());
064        return buf.getInt(20);
065    }
066
067    public void setValidLifetime(int validLifetime) {
068        ByteBuffer buf = ByteBuffer.wrap(getData());
069        buf.putInt(20, validLifetime);
070    }
071
072    @Override
073    public int getHeaderSize() {
074        return HEADER_SIZE;
075    }
076
077    @Override
078    public String toString() {
079        final StringBuilder values = new StringBuilder();
080        values.append("IP:");
081        values.append(getIp());
082        values.append(", Preferred-lifetime:");
083        values.append(getPreferredLifetime());
084        values.append(", Valid-lifetime:");
085        values.append(getValidLifetime());
086        values.append(", ");
087        try {
088            values.append(getOptions().toString());
089        } catch (DhcpException e) {
090            values.append("options:[");
091            values.append(toStringDataFallback(getOptionsRaw().array()));
092            values.append("]");
093        }
094
095        return getClass().getSimpleName() + "[" + getTagAsInt() + "]: " + values;
096    }
097
098    public static IaAddressOption create(@Nonnull final InetAddress ip, final int preferredLifetime, final int validaLifetime, final Optional<Dhcp6Options> options) {
099        final IaAddressOption iaNaOption = new IaAddressOption();
100        int length = HEADER_SIZE;
101
102        ByteBuffer encodedOptions = null;
103        if(options.isPresent() && !options.get().isEmpty()) {
104            encodedOptions = Dhcp6MessageEncoder.getInstance().encode(options.get());
105            length += encodedOptions.limit();
106        }
107
108        iaNaOption.setData(new byte[length]);
109        iaNaOption.setIp(ip);
110        iaNaOption.setPreferredLifetime(preferredLifetime);
111        iaNaOption.setValidLifetime(validaLifetime);
112        if (encodedOptions != null) {
113            iaNaOption.setOptions(encodedOptions);
114        }
115        return iaNaOption;
116    }
117}
