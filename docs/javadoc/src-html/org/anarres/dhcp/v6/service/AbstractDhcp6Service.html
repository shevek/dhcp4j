027/**
028 * Base DHCPv6 service outlining the standard DHCPv6 communication patterns defined in https://tools.ietf.org/html/rfc3315
029 *
030 * Created by marosmars.
031 */
032public abstract class AbstractDhcp6Service implements Dhcp6Service {
033
034    private static final Logger LOG = LoggerFactory.getLogger(LeaseManagerDhcp6Service.class);
035
036    private final Optional<ServerUnicastOption> serverUnicastOption;
037

047        this.serverUnicastOption = Optional.of(ServerUnicastOption.create(serverUnicastAddress));;

050    @Override
051    public Optional<Dhcp6Message> getReplyFor(@Nonnull Dhcp6RequestContext requestContext, @Nonnull final Dhcp6Message incomingMsg)
052            throws Dhcp6Exception {
053        Dhcp6Message reply;
054
055        switch (incomingMsg.getMessageType()) {
056            // TODO check reply types. Lease manager has the flexibility to change them
057            case DHCP_SOLICIT: {
058                Dhcp6Exception.InvalidMsgException.checkSolicit(incomingMsg);
059                reply = advertise(requestContext, incomingMsg);
060
061                // Add serverUnicastOption if configured
062                if (reply != null && unicastAllowed()) {
063                    reply.getOptions().add(serverUnicastOption.get());
064                }
065                break;
066            }
067            case DHCP_REQUEST: {
068                // TODO is this check correct ? if we dont allow unicasts, we check if we received from link local addr
069                // https://tools.ietf.org/html/rfc3315#section-18.2.1 first paragraph
070                if (!unicastAllowed() && !isLinkLocal(requestContext)) {
071                    reply = rejectUnicast(requestContext, incomingMsg);
072                } else {
073                    Dhcp6Exception.InvalidMsgException.checkRequest(incomingMsg, getServerId().getDuid());
074                    reply = reply(requestContext, incomingMsg);
075                }
076                break;
077            }
078            case DHCP_RENEW: {
079                // TODO is this check correct ? if we dont allow unicasts, we check if we received from link local addr
080                if (!unicastAllowed() && !isLinkLocal(requestContext)) {
081                    reply = rejectUnicast(requestContext, incomingMsg);
082                } else {
083                    Dhcp6Exception.InvalidMsgException.checkRenew(incomingMsg, getServerId().getDuid());
084                    reply = renew(requestContext, incomingMsg);
085                }
086                break;
087            }
088            case DHCP_REBIND: {
089                Dhcp6Exception.InvalidMsgException.checkRebind(incomingMsg);
090                reply = rebind(requestContext, incomingMsg);
091                break;
092            }
093            case DHCP_RELEASE: {
094                // TODO is this check correct ? if we dont allow unicasts, we check if we received from link local addr
095                if (!unicastAllowed() && !isLinkLocal(requestContext)) {
096                    reply = rejectUnicast(requestContext, incomingMsg);
097                } else {
098                    Dhcp6Exception.InvalidMsgException.checkRelease(incomingMsg, getServerId().getDuid());
099                    reply = release(requestContext, incomingMsg);
100                }
101                break;
102            }
103            case DHCP_CONFIRM: {
104                Dhcp6Exception.InvalidMsgException.checkConfirm(incomingMsg);
105                reply = confirm(requestContext, incomingMsg);
106                if (reply == null) {
107                    LOG.warn("Unable to confirm request: {}", incomingMsg);
108                }
109                break;
110            }
111            case DHCP_DECLINE: {
112                // TODO is this check correct ? if we dont allow unicasts, we check if we received from link local addr
113                if (!unicastAllowed() && !isLinkLocal(requestContext)) {
114                    reply = rejectUnicast(requestContext, incomingMsg);
115                } else {
116                    Dhcp6Exception.InvalidMsgException.checkDecline(incomingMsg, getServerId().getDuid());
117                    reply = decline(requestContext, incomingMsg);
118                }
119                break;
120            }
121            case DHCP_INFORMATION_REQUEST: {
122                Dhcp6Exception.InvalidMsgException.checkInformationRequest(incomingMsg, getServerId().getDuid());
123                reply = requestInformation(requestContext, incomingMsg);
124                break;
125            }
126            case DHCP_RELAY_FORW: {
127                Dhcp6Exception.InvalidMsgException.checkRelayForward(incomingMsg);
128                RelayMessageOption innerMsg = incomingMsg.getOptions().get(RelayMessageOption.class);
129                requestContext = new Dhcp6RequestContext(((Dhcp6RelayMessage) incomingMsg).getLinkAddress(), requestContext.getClientAddress(), incomingMsg.getOptions());
130                // Recursively call this method to get a reply to inner DHCP message. Just add replay msg options to request context
131                final Optional<Dhcp6Message> nestedReply = getReplyFor(requestContext.withRelayedOptions(incomingMsg.getOptions()), innerMsg.getRelayedMessage());
132
133                if (nestedReply.isPresent()) {
134                    reply = wrapAsRelayReply((Dhcp6RelayMessage) incomingMsg, nestedReply);
135                } else {
136                    reply = nestedReply.orNull();
137                }
138                break;
139            }
140
141            // https://tools.ietf.org/html/rfc3315#section-15.10
142            case DHCP_REPLY: {
143            }
144            // https://tools.ietf.org/html/rfc3315#section-15.11
145            case DHCP_RECONFIGURE: {
146            }
147            // https://tools.ietf.org/html/rfc3315#section-15.14
148            case DHCP_RELAY_REPL: {
149            }
150            // https://tools.ietf.org/html/rfc3315#section-15.3
151            case DHCP_ADVERTISE: {
152                // fall-through intentional, following messages are unexpected to be received on server side
153                throw new Dhcp6Exception.InvalidMsgException(incomingMsg.getMessageType());
154            }
155            default: {
156                reply = handle(requestContext, incomingMsg);
157            }
158        }
159
160        // Strip options to provide only requested
161        final OptionRequestOption optionRequestOption = incomingMsg.getOptions().get(OptionRequestOption.class);
162        if (reply != null && optionRequestOption != null) {
163            stripOptions(reply, optionRequestOption);
164        }
165
166        return Optional.fromNullable(reply);
167    }
168
169    protected final Dhcp6Message createReply(final Dhcp6Message incomingMsg) {
170        final Dhcp6Message reply = createGenericReply(incomingMsg, Dhcp6MessageType.DHCP_REPLY);
171        reply.getOptions().add(incomingMsg.getOptions().get(ClientIdOption.class));
172        return reply;
173    }
174
175    protected final Dhcp6Message createGenericReply(final Dhcp6Message incomingMsg, final Dhcp6MessageType type) {
176        final Dhcp6Message reply = new Dhcp6Message();
177        reply.setTransactionId(incomingMsg.getTransactionId());
178        reply.setMessageType(type);
179
180        final Dhcp6Options options = new Dhcp6Options();
181        options.add(getServerId());
182        reply.setOptions(options);
183        return reply;
184    }
185
186    private static boolean isLinkLocal(final @Nonnull Dhcp6RequestContext requestContext) {
187        return requestContext.getClientAddress().isLinkLocalAddress();
188    }
189
190    private static void stripOptions(final Dhcp6Message reply, final OptionRequestOption optionRequestOption) {
191        final HashSet<Short> requestedOptions = Sets.newHashSet(optionRequestOption.getRequestedOptions());
192
193        final Dhcp6Options filtered = new Dhcp6Options();
194
195        filtered.addAll(Iterables.filter(reply.getOptions(), new Predicate<Dhcp6Option>() {
196            @Override
197            public boolean apply(Dhcp6Option input) {
198                // keep only requested options + the standard options RFC3315
199                return input.getTag() <= 20 || requestedOptions.contains(input.getTag());
200            }
201        }));
202
203        reply.setOptions(filtered);
204    }
205
206    private boolean unicastAllowed() {
207        return serverUnicastOption.isPresent();
208    }
209
210    private Dhcp6RelayMessage wrapAsRelayReply(final @Nonnull Dhcp6RelayMessage incomingMsg,
211            final Optional<Dhcp6Message> nestedReply) {
212        final Dhcp6RelayMessage dhcp6RelayReply = new Dhcp6RelayMessage();
213        dhcp6RelayReply.setMessageType(Dhcp6MessageType.DHCP_RELAY_REPL);
214        dhcp6RelayReply.setHopCount(incomingMsg.getHopCount());
215        dhcp6RelayReply.setLinkAddress(incomingMsg.getLinkAddress());
216        dhcp6RelayReply.setPeerAddress(incomingMsg.getPeerAddress());
217        final Dhcp6Options options = new Dhcp6Options();
218        options.add(RelayMessageOption.create(nestedReply.get()));
219
220        // https://tools.ietf.org/html/rfc3315#section-22.18 - Copy interfaceId option if present in request
221        InterfaceIdOption interfaceIdOption = incomingMsg.getOptions().get(InterfaceIdOption.class);
222        if (interfaceIdOption != null)
223            options.add(interfaceIdOption);
224
225        dhcp6RelayReply.setOptions(options);
226        return dhcp6RelayReply;
227    }
228
229    /**
230     * Construct a reply message with status USE_MULTICAST
231     */
232    private Dhcp6Message rejectUnicast(final Dhcp6RequestContext requestContext, final Dhcp6Message incomingMsg) {
233        final ClientIdOption option = incomingMsg.getOptions().get(ClientIdOption.class);
234
235        LOG.warn("Rejecting message: {} from UNICAST: {} by client: {}", incomingMsg, requestContext.getClientAddress(),
236                (option == null) ? null : option.getDuid());
237
238        Dhcp6Message reply = new Dhcp6Message();
239        reply.setTransactionId(incomingMsg.getTransactionId());
240        reply.setMessageType(Dhcp6MessageType.DHCP_REPLY);
241
242        final Dhcp6Options options = new Dhcp6Options();
243
244        options.add(option);
245        options.add(getServerId());
246        options.add(StatusCodeOption.create(StatusCodeOption.USE_MULTICAST));
247        reply.setOptions(options);
248        return reply;
249    }
