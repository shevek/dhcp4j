020package org.apache.directory.server.dhcp.service.store;
021
022import com.google.common.base.Objects;
023import com.google.common.cache.Cache;
024import com.google.common.cache.CacheBuilder;
025import com.google.common.net.InetAddresses;
026import java.net.InetAddress;
027import java.util.ArrayList;
028import java.util.List;
029import java.util.concurrent.TimeUnit;
030import javax.annotation.CheckForNull;
031import javax.annotation.Nonnull;
032import org.anarres.dhcp.common.address.InterfaceAddress;
033import org.anarres.dhcp.common.address.NetworkAddress;
034import org.apache.directory.server.dhcp.DhcpException;
035import org.apache.directory.server.dhcp.io.DhcpRequestContext;
036import org.apache.directory.server.dhcp.messages.DhcpMessage;
037import org.apache.directory.server.dhcp.messages.HardwareAddress;
038import org.apache.directory.server.dhcp.messages.MessageType;
039import org.apache.directory.server.dhcp.options.vendor.SubnetMask;
040import org.apache.directory.server.dhcp.service.manager.AbstractLeaseManager;
041
042/**
043 * Very simple dummy/proof-of-concept implementation of a DhcpStore.
044 *
045 * @author <a href="mailto:dev@directory.apache.org">Apache Directory Project</a>
046 */
047public class SimpleStoreLeaseManager extends AbstractLeaseManager {
048    // private static final String DEFAULT_INITIAL_CONTEXT_FACTORY =
049    // "org.apache.directory.server.core.jndi.CoreContextFactory";
050
051    // a map of current leases
052    private final List<DhcpConfigSubnet> subnets = new ArrayList<DhcpConfigSubnet>();
053    private final Cache<HardwareAddress, Lease> leases = CacheBuilder.newBuilder()
054            .expireAfterAccess(TTL_LEASE.maxLeaseTime * 2, TimeUnit.SECONDS)
055            .recordStats()
056            .build();
057
058    //This will suppress PMD.AvoidUsingHardCodedIP warnings in this class
059    @SuppressWarnings("PMD.AvoidUsingHardCodedIP")
060    public SimpleStoreLeaseManager() {
061        subnets.add(new DhcpConfigSubnet(
062                NetworkAddress.forString("192.168.168.0/24"),
063                InetAddresses.forString("192.168.168.159"), InetAddresses.forString("192.168.168.179")));
064    }
065
066    /**
067     * Finds the subnet for the given client address.
068     */
069    @CheckForNull
070    protected DhcpConfigSubnet findSubnet(@Nonnull DhcpRequestContext context) {
071        for (InterfaceAddress localAddress : context.getInterfaceAddresses()) {
072            for (DhcpConfigSubnet subnet : subnets) {
073                if (subnet.contains(localAddress.getAddress()))
074                    return subnet;
075            }
076        }
077        return null;
078    }
079
080    @Override
081    public DhcpMessage leaseOffer(
082            DhcpRequestContext context,
083            DhcpMessage request,
084            InetAddress clientRequestedAddress, long clientRequestedExpirySecs)
085            throws DhcpException {
086        HardwareAddress hardwareAddress = request.getHardwareAddress();
087        Lease lease = leases.getIfPresent(hardwareAddress);
088        if (lease != null) {
089            lease.setState(Lease.LeaseState.OFFERED);
090            return newReply(request, MessageType.DHCPOFFER, lease);
091        }
092
093        DhcpConfigSubnet subnet = findSubnet(context);
094        if (subnet == null)
095            return null;
096
097        long leaseTimeSecs = getLeaseTime(TTL_OFFER, clientRequestedExpirySecs);
098
099        // TODO: Allocate a new address.
100        lease = new Lease(hardwareAddress, clientRequestedAddress);
101        lease.setState(Lease.LeaseState.OFFERED);
102        lease.setExpires(System.currentTimeMillis() / 1000 + leaseTimeSecs);
103        lease.getOptions().setAddressOption(SubnetMask.class, subnet.getNetwork().getNetmaskAddress());
104        leases.put(hardwareAddress, lease);
105
106        return newReply(request, MessageType.DHCPOFFER, lease);
107    }
108
109    @Override
110    public DhcpMessage leaseRequest(
111            DhcpRequestContext context,
112            DhcpMessage request,
113            InetAddress clientRequestedAddress, long clientRequestedExpirySecs) throws DhcpException {
114        HardwareAddress hardwareAddress = request.getHardwareAddress();
115        Lease lease = leases.getIfPresent(hardwareAddress);
116        if (lease == null)
117            return null;
118        if (!Objects.equal(lease.getClientAddress(), clientRequestedAddress))
119            return null;
120
121        lease.setState(Lease.LeaseState.ACTIVE);
122        long leaseTimeSecs = getLeaseTime(TTL_LEASE, clientRequestedExpirySecs);
123        lease.setExpires(System.currentTimeMillis() / 1000 + leaseTimeSecs);
124
125        return newReply(request, MessageType.DHCPACK, lease);
126    }
127
128    @Override
129    public boolean leaseDecline(
130            DhcpRequestContext context,
131            DhcpMessage request,
132            InetAddress clientAddress) throws DhcpException {
133        Lease lease = leases.asMap().remove(request.getHardwareAddress());
134        return lease != null
135                && Objects.equal(lease.getClientAddress(), clientAddress);
136    }
137
138    @Override
139    public boolean leaseRelease(
140            DhcpRequestContext context,
141            DhcpMessage request,
142            InetAddress clientAddress) throws DhcpException {
143        Lease lease = leases.asMap().remove(request.getHardwareAddress());
144        return lease != null
145                && Objects.equal(lease.getClientAddress(), clientAddress);
146    }
147}
