020package org.apache.directory.server.dhcp.options;
021
022import com.google.common.net.InetAddresses;
023import java.net.Inet4Address;
024import java.net.InetAddress;
025import java.net.UnknownHostException;
026import java.util.Arrays;
027import javax.annotation.Nonnull;
028import org.anarres.dhcp.common.address.AddressUtils;
029import org.apache.directory.server.dhcp.DhcpException;
030
031/**
032 * The Dynamic Host Configuration Protocol (DHCP) provides a framework
033 * for passing configuration information to hosts on a TCP/IP network.
034 * Configuration parameters and other control information are carried in
035 * tagged data items that are stored in the 'options' field of the DHCP
036 * message. The data items themselves are also called "options."
037 *
038 * This abstract base class is for options that carry a single IP address.
039 *
040 * @author <a href="mailto:dev@directory.apache.org">Apache Directory Project</a>
041 */
042public abstract class AddressOption extends DhcpOption {
043
044    @Nonnull
045    public Inet4Address getAddress() throws DhcpException {
046        try {
047            return (Inet4Address) Inet4Address.getByAddress(getData());
048        } catch (ClassCastException e) {
049            throw new DhcpException("Illegal InetAddress data: " + Arrays.toString(getData()) + " for " + this, e);
050        } catch (UnknownHostException e) {
051            throw new DhcpException("Illegal InetAddress data: " + Arrays.toString(getData()) + " for " + this, e);
052        }
053    }
054
055    public void setAddress(@Nonnull Inet4Address address) {
056        setData(address.getAddress());
057    }
058
059    /**
060     * @throws ClassCastException if the argument is not an Inet4Address.
061     */
062    public void setAddress(@Nonnull InetAddress address) {
063        setAddress((Inet4Address) address);
064    }
065
066    protected void validate(boolean allowZeroAddress) throws DhcpException {
067        super.validate();
068        if (getData().length != 4)
069            throw new DhcpException("Expected exactly 4 data bytes in " + this);
070        InetAddress address = getAddress();
071        if (!allowZeroAddress)
072            if (AddressUtils.isZeroAddress(address))
073                throw new DhcpException("Zero address not allowed in " + this);
074    }
075
076    @Override
077    public void validate() throws DhcpException {
078        validate(true);
079    }
080
081    @Override
082    protected String toStringData() throws DhcpException {
083        return InetAddresses.toAddrString(getAddress());
084    }
085
086}
