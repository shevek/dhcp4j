020package org.apache.directory.server.dhcp.options.dhcp;
021
022import javax.annotation.Nonnull;
023import org.apache.directory.server.dhcp.messages.MessageType;
024import org.apache.directory.server.dhcp.options.ByteOption;
025
026/**
027 * This option is used to convey the type of the DHCP message.  The code
028 * for this option is 53, and its length is 1.  Legal values for this
029 * option are:
030 * 
031 *         Value   Message Type
032 *         -----   ------------
033 *           1     DHCPDISCOVER
034 *           2     DHCPOFFER
035 *           3     DHCPREQUEST
036 *           4     DHCPDECLINE
037 *           5     DHCPACK
038 *           6     DHCPNAK
039 *           7     DHCPRELEASE
040 *           8     DHCPINFORM
041 * 
042 * @author <a href="mailto:dev@directory.apache.org">Apache Directory Project</a>
043 */
044public class DhcpMessageType extends ByteOption {
045
046    public DhcpMessageType() {
047    }
048
049    public DhcpMessageType(MessageType type) {
050        setMessageType(type);
051    }
052
053    /*
054     * @see org.apache.directory.server.dhcp.options.DhcpOption#getTag()
055     */
056    @Override
057    public byte getTag() {
058        return 53;
059    }
060
061    @Nonnull
062    public MessageType getMessageType() {
063        return MessageType.forTypeCode((byte) getByteValue());
064    }
065
066    public void setMessageType(MessageType type) {
067        setByteValue(type.getCode());
068    }
069
070    @Override
071    public String toStringData() {
072        return String.valueOf(getMessageType());
073    }
074}
